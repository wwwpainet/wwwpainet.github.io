                                # 3-字符串中每个单词的首字母大写

                                # 问题描述

                                是不是有时候会出现将一个字符串中的每个单词的首字母大写化，其余字符均为小写的情况，如"hello world"转化为"Hello World"。

                                # 问题分析

                                • 使用split()方法将字符串分解为单词数组
                                • 使用toLowerCase()方法将数组中的每一项均小写化
                                • 使用replace() 以及 toUpperCase()方法将数组中的每一项的首字母大写化，之后使用join()方法组合成一个新的字符串

                                # 解决方案

                                # (1) for + replace()

                                function titleCase(str) {
  //将字符串分解为数组并将其小写化
  var convertToArray = str.toLowerCase().split(" "); // [ 'hello', 'world' ]
  for (var i = 0; i < convertToArray.length; i++) {
    var char = convertToArray[i].charAt(0);  // h w
    //使用 replace()方法将数组中的每个首字母大写化
    convertToArray[i] = convertToArray[i].replace(char, function(char) {
      return char.toUpperCase();
    })
  }
  return convertToArray.join(" ");
}
console.log(titleCase("hello worLd"));   // Hello World

                                # (2) map() + replace()

                                function titleCase(str) {
  var convertToArray = str.toLowerCase().split(" ");
  var result = convertToArray.map(function(val) {
    return val.replace(val.charAt(0), val.charAt(0).toUpperCase());
  })
  return result.join(" ");
}

                                # (3) for ··· in + replace()

                                大家是否还记得数组遍历中的for ··· in方法，，是滴，这里也可以使用，如下所示:

                                function titleCases (str) {
    str = str.toLowerCase().split(" ");
    for (i in str) {
        str[i] = str[i].replace(str[i].charAt(0), str[i].charAt(0).toUpperCase());
    }
    return str.join(" ");
}

console.log(titleCases("hello world"));  // Hello World

                                # (4) map() + slice()

                                这里也可以不使用replace()方法，而是使用slice()方法进行大写化的首字母与其余字符串的拼接，如下所示:

                                function titleCases (str) {
    return str.toLowerCase().split(" ").map(function(word) {
        return (word.charAt(0).toUpperCase() + word.slice(1));
    }).join(" ");
}

console.log(titleCases("hello world"));

                                # (5) 正则匹配替换

                                ####思路分解: 利用正则表达式匹配出每一个单词，其实类似于字符串的数组分解 使用replace() 以及substr()方法进行首字母大写替换以及其余字符的小写化

                                function titleCases (str) {
    return str.replace(/\w\S*/g, function(word) {
        return word.charAt(0).toUpperCase() + word.slice(1).toLowerCase();
    })
}

console.log(titleCases("I'm a little tea pot", ""));  // I'm A Little Tea Pot

                                # (6) ES6解决方案

                                # 思路分解:

                                • /( |^)[a-z]/g - 匹配出整个字符串最开始的首字母以及以空格分隔的每个单词的首字母。
                                • 使用replace()方法进行首字母大写化替换。
                                function titleCases (str) {
    return str.toLowerCase().replace(/( |^)[a-z]/g, (L) => L.toUpperCase());
}

console.log(titleCases("I'm a little tea pot", ""));  //I'm A Little Tea Pot

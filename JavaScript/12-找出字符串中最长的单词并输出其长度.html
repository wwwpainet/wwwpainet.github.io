                                # 12-找出字符串中最长的单词并输出其长度

                                # 找出字符串中最长的单词并且将其长度输出。

                                这个算法其实就是让我们看看字符串中有多少个词，每个词有多少个字母，然后对这些词进行比较，找出字母数最多的那个词，并且返回这个最长字符数单词的长度。

                                实现上面描述的功能，同样先创建一个函数，比如findLongestWord()，并且给这个函数传入一个arr参数，这个arr是一个包含多个词的字符串。

                                然后在函数通过以下几步来实现所需的功能

                                • 先把字符串arr转为数组
                                • 将数组中的每个元素长度转换成一个新的数组
                                • 将这个数组按由小到大排序
                                • 取此数组中最后的数值，也就是最长的字符串
                                • 将这个长度值返回

                                # 将会大致用到下面几个有关于JavaScript点：

                                • for
                                • String.prototype.split() // 主要是用来将字符串str转换成数组arr。
                                • Array.prototype.sort()
                                • Array.prototype.reduce()
                                • Math.max()
                                • Array.prototype.pop()

                                # 方法

                                下面提供了几种实现findLongestWord(str)的方法，

                                不过这几种方法简单的可以分为for循环的、sort()和reduce()的。

                                function findLongestWord(arr) {
    // 第1步：将传给str的值"May the force be with you"
    var array = arr.split(' '); // 得到数组 ["May", "the", "force", "be", "with", "you"]
    var longest = 0;

    // 第2步：对数组array做遍历，并且将符合条件的值赋值给longest
    for (var i = 0, len = array.length; i < len; i++) {
        if (array[i].length > longest) {
            longest = array[i].length;
        }
    }
    return longest;
}
console.log(findLongestWord("The quick brown fox jumped over the lazy dog"));  // 6
console.log(findLongestWord("May the force be with you"));  // 5

                                # 还可以for循环之后，通过Math.max()直接将数组arrNum中的最大值取出。如下所示：

                                function findLongestWord (str) {
    // 第1步：将传给str的值"May the force be with you"
    var arr = str.split(' ');
    var arrNum = [];

    for (var i = 0, len = arr.length; i < len; i++) {
        // 将数组arr中每个元素的长度length放到一个新数组arrNum中
        arrNum.push(arr[i].length);
    }

    // 第3步： 通过Math.max()取出数组arrNum中最大值，并且输出
    return Math.max.apply(null, arrNum);
}

console.log(findLongestWord("The quick brown fox jumped over the lazy dog"));  // 6
console.log(findLongestWord("May the force be with you"));  // 5

                                # sort()方法

                                function findLongestWord (str) {
    var arr = str.split(' ');
    // 第2步： 通过sort()将数组arr按由小到大排序
    arr.sort(function(a, b) {
        return a.length - b.length;
    })
    // 最长的字符排在数组最后
    // 第3步：获取数组最长字符的长度
    var longestString = arr.pop().length;
    // 第4步：返回最长字符的长度
    return longestString;
}

console.log(findLongestWord("The quick brown fox jumped over the lazy dog"));  // 6
console.log(findLongestWord("May the force be with you"));  // 5

                                # reduce()方法

                                function findLongestWord (str) {
    var strSplit = str.split(' ');
    // 第2步使用reduce方法，取到strSplit数组中最长的元素
    var longestWord = strSplit.reduce(function(longest, currentWord) {
        return currentWord.length > longest.length ? currentWord : longest;
    }, "");
    // 取到最长的元素longestWord
    // 第3步. 返回longestWord的length
    return longestWord.length;
}

console.log(findLongestWord("The quick brown fox jumped over the lazy dog"));  // 6
console.log(findLongestWord("May the force be with you"));  // 5

                                # reduce()方法和Math.max()方法配合使用:

                                function findLongestWord (str) {
    var strSplit = str.split(' ');
    return strSplit.reduce(function(longest, currentWord) {
        // 第2步使用reduce方法，取到strSplit数组中最长的元素的length，并且返回
        return Math.max(longest, currentWord.length);
    }, 0);
}

console.log(findLongestWord("The quick brown fox jumped over the lazy dog"));  // 6
console.log(findLongestWord("May the force be with you"));  // 5
